================================================================================
                        COMMON LINQ SNIPPETS
================================================================================

1️⃣ BASIC GET WITH SOFT DELETE:
   await _context.Entity
       .Where(e => e.Id == id && !e.IsDeleted)
       .FirstOrDefaultAsync();

2️⃣ INCLUDE SINGLE RELATIONSHIP:
   await _context.Doctor
       .Include(d => d.Department)
       .Where(d => d.Id == id && !d.IsDeleted)
       .FirstOrDefaultAsync();

3️⃣ INCLUDE WITH NESTED THENINCLUDE:
   await _context.Session
       .Include(s => s.Doctor)
           .ThenInclude(d => d.Department)
       .Include(s => s.Patient)
       .Where(s => s.Id == id && !s.IsDeleted)
       .FirstOrDefaultAsync();

4️⃣ INCLUDE WITH COLLECTION FILTERING:
   await _context.Department
       .Include(d => d.Doctors.Where(doc => !doc.IsDeleted))
       .Where(d => d.Id == id && !d.IsDeleted)
       .FirstOrDefaultAsync();

5️⃣ FILTER BY FOREIGN KEY:
   await _context.Doctor
       .Where(d => d.DepartmentId == departmentId && !d.IsDeleted)
       .OrderBy(d => d.Name)
       .ToListAsync();

6️⃣ FILTER BY ENUM:
   await _context.RequestLabs
       .Where(rl => rl.Status == LabRequestStatus.Pending && !rl.IsDeleted)
       .OrderByDescending(rl => rl.RequestedAt)
       .ToListAsync();

7️⃣ FILTER BY MULTIPLE CONDITIONS:
   await _context.Notification
       .Where(n => n.UserId == userId && !n.IsRead && !n.IsDeleted)
       .OrderByDescending(n => n.SentAt)
       .ToListAsync();

8️⃣ COMPOSITE KEY QUERY:
   await _context.PatientResultElement
       .Where(pre => pre.PatientResultId == patientResultId && 
                    pre.TestElementId == testElementId && 
                    !pre.IsDeleted)
       .FirstOrDefaultAsync();

9️⃣ EXISTENCE CHECK (Efficient):
   await _context.Doctor
       .AnyAsync(d => d.Id == id && !d.IsDeleted);

🔟 COLLECTION INCLUDE WITH NESTED RELATIONS:
   await _context.PatientResult
       .Include(pr => pr.ResultElements)
           .ThenInclude(pre => pre.TestElement)
       .Include(pr => pr.ResultElements)
           .ThenInclude(pre => pre.Technician)
       .Where(pr => pr.Id == id && !pr.IsDeleted)
       .FirstOrDefaultAsync();

================================================================================
                        IMPLEMENTATION FEATURES
================================================================================

✅ Consistent Async/Await Pattern
   - All methods are async Task<T> or Task<bool>
   - Uses SaveChangesAsync() for database operations

✅ Soft Delete Strategy
   - WHERE !IsDeleted in all queries
   - SoftDeleteAsync sets flag instead of removing
   - DeleteAsync performs hard delete when needed

✅ Timestamp Management
   - CreatedAt set on Add
   - UpdatedAt set on Update/SoftDelete
   - Uses DateTime.UtcNow for consistency

✅ Comprehensive Comments
   - Each method has XML documentation
   - Complex queries explain the JOIN structure
   - Operation purpose is clear

✅ Null Safety
   - Nullable return types: T?
   - Null coalescing where appropriate
   - Entity validation before operations

✅ Exception Handling
   - MarkAsReadAsync throws if notification not found
   - Delete methods return bool for graceful handling
   - Null checks prevent NullReferenceExceptions

================================================================================
                      HOW TO USE IN CONTROLLERS
================================================================================

Example 1: Simple CRUD
    public class DoctorController
    {
        private readonly IDoctorRepo _doctorRepo;

        public DoctorController(IDoctorRepo doctorRepo)
        {
            _doctorRepo = doctorRepo;
        }

        [HttpGet("{id}")]
        public async Task<IActionResult> GetDoctor(int id)
        {
            var doctor = await _doctorRepo.GetByIdAsync(id);
            if (doctor == null)
                return NotFound();
            return Ok(doctor);
        }
    }

Example 2: Complex Query with Relationships
    [HttpGet("{id}/details")]
    public async Task<IActionResult> GetSessionDetails(int id)
    {
        var session = await _sessionRepo.GetWithDetailsAsync(id);
        // Now session includes: Patient, Doctor (with Department), Department
        return Ok(session);
    }

Example 3: Filtering
    [HttpGet("patient/{patientId}")]
    public async Task<IActionResult> GetPatientSessions(int patientId)
    {
        var sessions = await _sessionRepo.GetByPatientAsync(patientId);
        return Ok(sessions);
    }

Example 4: Notifications
    [HttpGet("unread")]
    public async Task<IActionResult> GetUnreadNotifications(int userId)
    {
        var notifications = await _notificationRepo.GetUnreadByUserAsync(userId);
        return Ok(notifications);
    }
